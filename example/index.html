jQuery Form Validator v2.0

This is a form validation (jQuery) plugin with a rich set of features. The goal of the plugin is to keep your html clean from javascript logic and at the same time being as light weight as possible.

<?php
session_start();
if( isset($_POST['captcha']) && isset($_SESSION['captcha'])) {
  if($_POST['captcha'] != ($_SESSION['captcha'][0]+$_SESSION['captcha'][1]))
    die('Invalid captcha answer');  // client does not have javascript enabled
}

$_SESSION['captcha'] = array( mt_rand(0,9), mt_rand(1, 9) );
$captcha = $_SESSION['captcha'][0] + $_SESSION['captcha'][1];

?><html>
<head>

                $('#pass').displayPasswordStrength();

check.php

<?php
$json = array();

if(count($_POST) == 0)
    $json['message'] = 'No data posted';

elseif(empty($_POST['validate']))
    $json['message'] = 'Validate argument was empty';

else {
    $user_model = new User_Model();
    if($user_model->loadUserByEmail($_POST['validate']))
        $json['message'] = 'E-mail address already registered';
    else
        $json['success'] = true;
}

header('Content-Type: application/json');
echo json_encode($json);
