jQuery Form Validator v2.0

This is a form validation (jQuery) plugin with a rich set of features. The goal of the plugin is to keep your html clean from javascript logic and at the same time being as light weight as possible.

Module: security

<?php
session_start();
if( isset($_POST['captcha']) && isset($_SESSION['captcha'])) {
  if($_POST['captcha'] != ($_SESSION['captcha'][0]+$_SESSION['captcha'][1]))
    die('Invalid captcha answer');  // client does not have javascript enabled
}

$_SESSION['captcha'] = array( mt_rand(0,9), mt_rand(1, 9) );

?><html>
<head>
    <script src="js/formvalidator/jquery.formvalidator.min.js"></script>
    <script>
        $.formUtils.loadModules('security');
    </script>
</head>

  <form action="" onsubmit="return $(this).validate();">
    <div>

        <input type="password" name="pass" id="pass"
                    data-validation="validate_strength strength1 validate_confirmation" />
        <script>
            $(function() {
                $('#pass').displayPasswordStrength();
            });
        </script>
    </div>
    <div>
        <label for="confirm_pass">Confirm password</label>
        <input type="password" name="pass_confirmation" id="confirm_pass"  />
    </div>

            What is the sum of <?=$_SESSION['captcha'][0]?> + <?=$_SESSION['captcha'][1]?>?

        <input type="text" name="captcha" id="cap" data-validation="validate_spamcheck captcha8" />
    </div>
    <div>

    </div>
